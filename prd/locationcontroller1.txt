const pool = require("../db");


//helper function 1 : title case 
function toTitleCase(str) {
    return str
        .toLowerCase()
        .replace(/\b\w/g, char => char.toUpperCase());
}




// helper function 2 : standard succes response
function sendResponse(res,data,startTime) {
    const responseTime = Date.now() - startTime;


    res.json({
        success :true,
        count : data.length,
        data : data,
        meta : {
            responseTime : responseTime
        }
    });
}

//1. get all states
exports.getStates = async (req, res) => {
    const startTime = Date.now();
    try{
        const result = await pool.query("SELECT * FROM states ORDER BY state_name");
        sendResponse(res,result.rows,startTime);
    } catch (err) {
        console.error(err);
        res.status(500).json({error:"server error"});
    }
}




//2.get all districts
exports.getDistricts = async (req, res) => {
    const startTime = Date.now();
    try{
        const {state_code} = req.params;

        const result = await pool.query(
            "SELECT d.* FROM districts d JOIN states s ON d.state_id = s.id WHERE s.state_code = $1 ORDER BY d.district_name",
            [Number(state_code)]
        );
        sendResponse(res,result.rows,startTime);
    } catch (err) {
        console.error(err);
        res.status(500).json({error:"server error"});
    }
}

//search api
exports.searchVillages = async (req,res) => {
    const startTime = Date.now();
    try {
        const { q } = req.query;

        const result = await pool.query(
            `SELECT  v.id,v.village_name,sd.subdistrict_name,d.district_name,s.state_name
            FROM villages v
            JOIN subdistricts sd 
            ON v.subdistrict_id = sd.id
            JOIN districts d 
            ON sd.district_id = d.id
            JOIN states s 
            ON d.state_id = s.id
            WHERE v.village_name ILIKE $1
            ORDER BY 
            (v.village_name ILIKE $2) DESC,
            v.village_name ASC
            LIMIT 20`,
            [`${q}%`,`${q}%`]
        );

       const formatted = result.rows.map(row => {
        const name = row.village_name
                .replace(/\(\d+\)/g, "")
                .replace(/([a-z])([A-Z])/g, "$1 $2")
                .replace(/\s+/g, " ")
                .replace(/\.\s*/g, ". ")
                .trim()
                .replace(/^./, c => c.toUpperCase());
        return {
            value : row.id,
            label : name,
            fullAddress : `${name}, ${row.subdistrict_name}, ${row.district_name}, ${toTitleCase(row.state_name)}, India`,
            hierarchy : {
                village : name,
                subDistrict : row.subdistrict_name,
                district : row.district_name,
                state : toTitleCase(row.state_name),
                country : "India"
            }
        };

       });
        sendResponse(res,formatted,startTime);
    } catch (err) {
        console.error(err);
        res.status(500).json({error:"server error"});
    }
}
